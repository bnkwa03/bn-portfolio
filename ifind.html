# Cluster 200+ open-ended responses into themes with an LLM
import anthropic, pandas as pd

df = pd.read_csv("interview_responses.csv")
client = anthropic.Anthropic()

def tag_theme(quote):
    msg = client.messages.create(
        model="claude-opus-4-8",
        max_tokens=60,
        messages=[{"role": "user",
                   "content": f"Tag the core need in: {quote}"}],
    )
    return msg.content[0].text

df["theme"] = df["response"].apply(tag_theme)
